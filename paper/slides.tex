\vspace{-2em}
\begin{block}{TL;DR --- a progress update}
\small Train a \textbf{tiny adapter} on a \textbf{frozen} audio--visual model to make it use audio more.
Na\"ive offline RL-DPO \textbf{collapses} it completely; a \textbf{two-part fix} prevents that. Honest result: the fix
works cleanly \emph{as a fix}, but the grounding gain is \textbf{small} and shows up on \textbf{two of three}

\end{block}

% --- 2. The problem ---
\begin{frame}{The problem: a known modality bias}
\begin{itemize}
\item AV-LLMs \textbf{over-trust what they see}, \textbf{under-use what they hear}.
\item They hallucinate a sound the image implies; they ignore audio that contradicts the video.
\end{itemize}
\vfill
\begin{center}\large\itshape
Narrow, practical question: fix this \textbf{without retraining} ---\\ freeze everything, bolt on something tiny?
\end{center}
\end{frame}

{\small Each adapter is a variational bottleneck on its token stream $x$:}
\vspace{-0.3em}
\[ z=\mu(x)+\sigma(x)\odot\epsilon,\ \ \epsilon\sim\mathcal N(0,I);\qquad y=x+W_{\text{out}}\,z \]

\item $W_{\text{out}}$ \textbf{zero-init} $\Rightarrow$ $y{=}x$ at start; \textbf{bypass} $\Rightarrow$ base predictions for free.
\end{itemize}

% --- 7. What broke ---
\begin{frame}{What broke: na\"ive DPO collapses the model}
\begin{itemize}
\item Standard DPO $\Rightarrow$ perception accuracy \textbf{0.95 $\to$ 0.1}; converges to a near-constant ``no''.
\end{itemize}
\vfill
\begin{center}
\setbeamercolor{block body}{bg=red!8}
\begin{block}{The trap}
Answering ``no'' to everything looks \textbf{perfect} on a hallucination metric --- you never affirm anything
false. \textbf{Wrong training metric!} 
\end{block}
\end{center}
\end{frame}

% --- Read the loss: trained vs frozen (place before "Why it broke") ---
\begin{frame}{Read the loss: what's trained, what's frozen}
\centering
\renewcommand{\arraystretch}{1.35}
\begin{tabular}{@{}r@{\quad}l@{}}
\textcolor{trn}{\textbf{trained}}        & $\theta$ = the two VIB adapters \ (inside: $\mu(x),\ \sigma(x)$) \\
\textcolor{trn}{\textbf{move with }$\theta$} & $c_p,\ r_p$ \ (policy log-probs), \quad $p_\theta$ \\
\textcolor{frz}{\textbf{frozen base}}     & $c_r,\ r_r$ \ (reference $=$ bypass), \quad $p_{\text{base}}$ \\
\textbf{fixed}                            & $\beta,\ \beta_{kl},\ \lambda_a,\ \lambda_{kl},\ \delta$ \ (hyperparams); \ the sigmoid in $-\log\sigma$ \\
\end{tabular}
\vfill
{\small Subscript $p$ = \textcolor{trn}{policy} (adapter \emph{on}) \ $\cdot$ \ subscript $r$ = \textcolor{frz}{reference} (adapter \emph{bypassed} $=$ frozen base).\\
Only $\theta$ is a learnable parameter --- the \textcolor{trn}{policy side} moves, the \textcolor{frz}{reference side} is constant.}
\end{frame}

% --- 8. Why it broke ---
\begin{frame}{Why it broke}
\[ \mathcal L_0=\underbrace{-\log\sigma\!\big(\beta[(c_p{-}c_r){-}(r_p{-}r_r)]\big)}_{\text{swap-DPO margin}}\;+\;\underbrace{\beta_{kl}\,\mathrm{KL}_{\text{VIB}}}_{\text{rate on the adapter only}} \]
\vspace{-0.4em}
\begin{itemize}\small
\item $c_p,r_p$ = policy log-probs of chosen/rejected letters; $c_r,r_r$ = the frozen base.
\item \textbf{Likelihood displacement}: the margin grows while \emph{both} $c_p$ and $r_p$ \emph{fall} $\Rightarrow$ the chosen answer gets \emph{less} likely.
\item $\mathrm{KL}_{\text{VIB}}$ limits the adapter's \emph{rate}, \textbf{not the output} $\Rightarrow$ nothing stops the drift to constant ``no''.
\end{itemize}
\end{frame}

% --- 9. The fix ---
\begin{frame}{The fix: two anchors}
{\small Add two terms to the na\"ive loss:}
\vspace{-0.3em}
\begin{align*}
\mathcal L =\ & \underbrace{-\log\sigma(\beta[(c_p{-}c_r){-}(r_p{-}r_r)])}_{\text{swap-DPO}}
 \;+\; \lambda_a\,\underbrace{\big({-}\log\sigma(\beta(c_p{-}c_r){-}\delta)\big)}_{\textcolor{trn}{\textbf{chosen anchor}}} \\[2pt]
 &+\; \lambda_{kl}\,\underbrace{\mathrm{KL}\!\big(p_{\text{base}}(\cdot|x)\,\|\,p_\theta(\cdot|x)\big)}_{\textcolor{trn}{\textbf{KL-to-base}}}
 \;+\; \beta_{kl}\,\mathrm{KL}_{\text{VIB}}
\end{align*}
\vspace{-0.5em}
\begin{itemize}\small
\item \textcolor{trn}{\textbf{chosen anchor}}: pins the chosen letter's log-prob $\ge$ frozen base $\Rightarrow$ protects \textbf{PA}.
\item \textcolor{trn}{\textbf{KL-to-base}}: on ordinary (non-swap) inputs, keeps the output near base $\Rightarrow$ protects \textbf{HR}.
\end{itemize}
\end{frame}

\begin{frame}{GRPO on a Frozen AV-LLM via a VIB Adapter}
\textbf{Setup.} The audio-visual LLM is \emph{frozen}; the only trainable
parameters are a tiny \emph{per-modality} Variational Information Bottleneck
(audio, vision) wrapping the modality-adapter outputs.
\[
  z = \mu(x) + \sigma(x)\odot\epsilon,\qquad
  y = x + W_{\text{out}}\,z,\qquad W_{\text{out}}\!\stackrel{\text{init}}{=}0
\]
$W_{\text{out}}$ zero-init $\Rightarrow$ identity at init (base for free);
\textbf{exploration comes from the reparam noise} $\sigma\odot\epsilon$.

\medskip
\textbf{Rollout \& reward (verifiable, no reward model).}
For a yes/no grounding question, draw a group of $G$ samples; the action is a
categorical over the candidate answer tokens $\{\text{yes},\text{no},(\text{abstain})\}$.
\[
  r_i = +1\ (\text{correct}),\quad 0\ (\text{abstain}),\quad -1\ (\text{hallucinate})
\]

\textbf{Group-relative advantage \& objective.}
\[
  A_i = \frac{r_i-\mathrm{mean}(r)}{\mathrm{std}(r)+\varepsilon},\qquad
  \mathcal{L} = -\tfrac{1}{G}\!\sum_i A_i \log\pi(a_i\mid x)
  \;+\;\beta_{\mathrm{kl}}\,\mathrm{KL}\!\big(q(z|x)\,\|\,\mathcal N(0,I)\big)
  \;+\;\lambda_{\mathrm{ref}}\,\mathrm{KL}(\text{base}\,\|\,\pi)
\]

\end{frame}

% ======================= SLIDE 2 =======================
\begin{frame}{Signal Collapse: Diagnosis, First Results, Remedy}
\textbf{The collapse.} A near-deterministic frozen policy $+$ an \emph{easy}
verifiable reward makes most groups \emph{unanimous} $\Rightarrow A_i\equiv 0
\Rightarrow$ the policy gradient vanishes (``KL-regularized RL is designed to
mode-collapse''). Only the KL terms move the adapter --- back toward identity.

\medskip
\textbf{First results} (Qwen3-Omni, AVE yes/no probe):
\begin{itemize}
  \item Strong base: probe acc $\approx 0.97$, $p_{\text{correct}}\approx 0.875$.
  \item $\mathrm{adv\_std}=0$ on $\sim\!2/3$ of steps $\Rightarrow$ \emph{no learning signal}.
  
\end{itemize}

\medskip
\textbf{Next lever:} harder \emph{counterfactual (swap)} items
        $\Rightarrow$ genuine uncertainty $\Rightarrow$ reward variance.

\end{frame}
