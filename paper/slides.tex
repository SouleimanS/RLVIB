% --- 1. Title + TL;DR ---
{\setbeamertemplate{footline}{}
\begin{frame}
\titlepage
\vspace{-2em}
\begin{block}{TL;DR --- a progress update}
\small Train a \textbf{tiny adapter} on a \textbf{frozen} audio--visual model to make it use audio more.
Na\"ive DPO \textbf{collapses} it completely; a \textbf{two-part fix} prevents that. Honest result: the fix
works cleanly \emph{as a fix}, but the grounding gain is \textbf{small} and shows up on \textbf{one of three}
models. The last third is decisions I want your input on.
\end{block}
\end{frame}}

% --- 2. The problem ---
\begin{frame}{The problem: a known modality bias}
\begin{itemize}
\item AV-LLMs \textbf{over-trust what they see}, \textbf{under-use what they hear}.
\item They hallucinate a sound the image implies; they ignore audio that contradicts the video.
\end{itemize}
\vfill
\begin{center}\large\itshape
Narrow, practical question: fix this \textbf{without retraining} ---\\ freeze everything, bolt on something tiny?
\end{center}
\end{frame}

% --- 3. The idea: architecture ---
\begin{frame}{The idea: freeze the model, train two tiny modules}
\centering

\vspace{0.2em}
{\small Each adapter is a variational bottleneck on its token stream $x$:}
\vspace{-0.3em}
\[ z=\mu(x)+\sigma(x)\odot\epsilon,\ \ \epsilon\sim\mathcal N(0,I);\qquad y=x+W_{\text{out}}\,z \]
\vspace{-0.7em}
\begin{itemize}\small
\item Same form for \textbf{audio and vision}; the \textbf{only} trainable parameters.
\item $W_{\text{out}}$ \textbf{zero-init} $\Rightarrow$ $y{=}x$ at start; \textbf{bypass} $\Rightarrow$ base predictions for free.
\end{itemize}
\end{frame}

% --- 4. Setup / what was actually run ---
\begin{frame}{Setup (so the numbers are concrete)}
\begin{columns}[T]
\begin{column}{0.5\textwidth}
\textbf{Training}
\begin{itemize}\small
\item Data: \textbf{300 audio-swapped AVE clips}, \textbf{2 epochs}.
\item Loss weights: $\beta{=}0.1$, $\lambda_a{=}1$, $\lambda_{kl}{=}2$, $\beta_{kl}{=}0.01$.
\item Only the two adapters update; backbone frozen.
\end{itemize}
\end{column}
\begin{column}{0.5\textwidth}
\textbf{Evaluation}
\begin{itemize}\small
\item \textbf{AVHBench} --- the target (AV hallucination).
\item \textbf{CMM} --- capability: \textbf{PA} + \textbf{HR}.
\item \textbf{DAVE} --- diagnostic (base only).
\item $300$-example subsets; \textbf{150 val} (select) / \textbf{150 test} (report); \textbf{paired} tests.
\end{itemize}
\end{column}
\end{columns}
\vfill
{\footnotesize\textbf{Backbones / encoders:} Qwen3-Omni (\emph{AuT} audio, Qwen2.5-VL vision) \ $\cdot$ \ Qwen2.5-Omni (\emph{Whisper-v3} audio, Qwen2.5-VL vision) \ $\cdot$ \ VideoLLaMA2 (\emph{BEATs} audio, \emph{SigLIP} vision).}
\end{frame}

% --- 5. The training signal (the core trick) ---
\begin{frame}{The core trick: manufacture an audio--vision conflict}
\centering
\begin{tikzpicture}[font=\small,>={Latex[length=1.8mm]},
  clip/.style={draw,rounded corners,fill=gray!12,align=center,minimum width=34mm,minimum height=17mm},
  good/.style={draw,rounded corners,fill=green!14,align=center,minimum width=46mm,minimum height=10mm},
  bad/.style={draw,rounded corners,fill=red!14,align=center,minimum width=46mm,minimum height=10mm}]
  \node[clip] (clip) {\textbf{audio-swapped clip}\\[1mm]\emph{see}: a dog\\\emph{hear}: a bell};
  \node[bad,right=16mm of clip,yshift=9mm] (bad) {``a dog barking'' \ {\scriptsize ignores audio} $\boldsymbol{\times}$};
  \node[good,right=16mm of clip,yshift=-9mm] (good) {``a bell'' \ {\scriptsize uses audio} $\boldsymbol{\checkmark}$};
  \draw[->] (clip.east) to[out=25,in=180] (bad.west);
  \draw[->] (clip.east) to[out=-25,in=180] (good.west);
\end{tikzpicture}
\vfill
\small Swap a clip's audio $\Rightarrow$ seen $\neq$ heard. Both answers are scored on the \emph{same} clip,
so preferring the \textbf{heard} answer is only possible by \textbf{attending to the audio}.
\end{frame}

% --- 6. What broke ---
\begin{frame}{What broke: na\"ive DPO collapses the model}
\begin{itemize}
\item Standard DPO $\Rightarrow$ perception accuracy \textbf{0.95 $\to$ 0.01}; converges to a near-constant ``no''.
\end{itemize}
\vfill
\begin{center}
\setbeamercolor{block body}{bg=red!8}
\begin{block}{The trap}
Answering ``no'' to everything looks \textbf{perfect} on a hallucination metric --- you never affirm anything
false. \textbf{Watch the wrong metric and the collapse is invisible.} (Back to this on the benchmarks slide.)
\end{block}
\end{center}
\end{frame}

% --- 7. Why it broke ---
\begin{frame}{Why it broke (a documented failure mode, not a bug)}
\[ \mathcal L_0=\underbrace{-\log\sigma\!\big(\beta[(c_p{-}c_r){-}(r_p{-}r_r)]\big)}_{\text{swap-DPO margin}}\;+\;\underbrace{\beta_{kl}\,\mathrm{KL}_{\text{VIB}}}_{\text{rate on the adapter only}} \]
\vspace{-0.4em}
\begin{itemize}\small
\item $c_p,r_p$ = policy log-probs of chosen/rejected letters; $c_r,r_r$ = the frozen base.
\item \textbf{Likelihood displacement}: the margin grows while \emph{both} $c_p$ and $r_p$ \emph{fall} $\Rightarrow$ the chosen answer gets \emph{less} likely.
\item $\mathrm{KL}_{\text{VIB}}$ limits the adapter's \emph{rate}, \textbf{not the output} $\Rightarrow$ nothing stops the drift to constant ``no''.
\end{itemize}
\end{frame}

% --- 8. The fix ---
\begin{frame}{The fix: two anchors}
{\small Add two terms to the na\"ive loss:}
\vspace{-0.3em}
\begin{align*}
\mathcal L =\ & \underbrace{-\log\sigma(\beta[(c_p{-}c_r){-}(r_p{-}r_r)])}_{\text{swap-DPO}}
 \;+\; \lambda_a\,\underbrace{\big({-}\log\sigma(\beta(c_p{-}c_r){-}\delta)\big)}_{\textcolor{trn}{\textbf{chosen anchor}}} \\[2pt]
 &+\; \lambda_{kl}\,\underbrace{\mathrm{KL}\!\big(p_{\text{base}}(\cdot|x)\,\|\,p_\theta(\cdot|x)\big)}_{\textcolor{trn}{\textbf{KL-to-base}}}
 \;+\; \beta_{kl}\,\mathrm{KL}_{\text{VIB}}
\end{align*}
\vspace{-0.5em}
\begin{itemize}\small
\item \textcolor{trn}{\textbf{chosen anchor}}: pins the chosen letter's log-prob $\ge$ frozen base $\Rightarrow$ protects \textbf{PA}.
\item \textcolor{trn}{\textbf{KL-to-base}}: on ordinary (non-swap) inputs, keeps the output near base $\Rightarrow$ protects \textbf{HR}.
\end{itemize}
\end{frame}

% --- 9. Ablation (strongest result) ---
\begin{frame}{Ablation (Qwen3-Omni): both anchors are necessary}
\centering
\begin{tabular}{lccc}
\toprule
objective & AVHBench & CMM-PA & CMM-HR \\
\midrule
frozen base            & 0.647 & 0.939 & 0.810 \\
na\"ive swap-DPO       & 0.673 & 0.939 & 0.774 \\
\;+ chosen anchor only & 0.660 & 0.939 & 0.798 \\
\;+ KL-to-base only    & 0.647 & 0.924 & 0.810 \\
\;+ \textbf{both}      & \textbf{0.707} & 0.909 & 0.869 \\
\bottomrule
\end{tabular}
\vfill
{\footnotesize\textbf{+both, AVHBench by type:} A$\to$V $0.65{\to}0.68$ \ $\cdot$ \ V$\to$A $0.72{\to}0.78$ \ $\cdot$ \ AV-match $0.54{\to}0.63$ \ (overall $0.647{\to}0.707$).}\\[2pt]
{\small Each \emph{alone} fails differently --- chosen-only PA$\to$0.30, KL-only HR$\to$0.66; \textbf{only both} stay healthy \emph{and} score best.}
\end{frame}

% --- 10. Benchmarks: what the numbers mean ---
\begin{frame}{What the metrics mean (and why the collapse hid)}
\begin{itemize}\small
\item \textbf{AVHBench} = the \textbf{target} (audio--visual hallucination).
\item \textbf{CMM} = capability, two axes:
  \begin{itemize}\scriptsize
  \item \textbf{PA} (perception): on \emph{present} items --- correctly says ``yes, it's there'' (catches under-perception).
  \item \textbf{HR} (resistance): on \emph{absent} items --- correctly says ``no, it isn't'' (catches hallucination).
  \end{itemize}
\end{itemize}
\vfill
\begin{block}{The collapse, in metric terms}
\small All-``no'' $\Rightarrow$ \textbf{HR$\to$1.0} but \textbf{PA$\to$0}: HR alone says ``perfect''. \textbf{Watch both} --- they map onto the anchors (chosen$\leftrightarrow$PA, KL$\leftrightarrow$HR).
\end{block}
\end{frame}

\begin{frame}{Generalization: uneven across backbones}
\centering
\begin{tabular}{lccc}
\toprule

% --- 15. What the module actually learned ---
\begin{frame}{The twist: what the module actually learned}
\centering
\begin{tikzpicture}[font=\small]
  \node[draw,rounded corners,fill=trn!22,minimum width=68mm,minimum height=11mm,align=center] (v)
    {\textbf{vision module}: rewrites $\sim$59\% of each vision token};
  \node[draw,rounded corners,fill=gray!12,minimum width=68mm,minimum height=11mm,align=center,below=4mm of v] (a)
    {\textbf{audio module}: $\approx$ pass-through (does almost nothing)};
\end{tikzpicture}
\begin{itemize}\small
\item ``Audio grounding'' is achieved by \textbf{reshaping the vision stream}; the frozen LM resolves the conflict.
\item The information bottleneck barely compresses at these settings.
\item $\Rightarrow$ two framing nouns (``\emph{audio}'', ``\emph{information bottleneck}'') don't describe what happens. A real finding --- how to frame it?
\end{itemize}
\end{frame}

% --- 16. Methodology I trust ---
\begin{frame}{Methodology (so you trust the numbers)}
\begin{itemize}\small
\item \textbf{Val/test split} per benchmark: choose checkpoints on val, report on a disjoint test half.
\item \textbf{Paired} significance tests (McNemar + bootstrap), not bare averages.
\item \textbf{Constant-answer detector} confirms the selected model didn't quietly collapse.
\item Caught + fixed a \textbf{parser bug} (scored one model's prose answers as errors); \textbf{no change} to the
Qwen models $\Rightarrow$ the fix is cleanly isolated.
\end{itemize}
\end{frame}
